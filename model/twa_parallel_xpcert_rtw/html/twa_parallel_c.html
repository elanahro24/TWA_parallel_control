  568     }

  571       eml_e_c = 6 * eml_i;
  572       for (eml_d_k = 0; eml_d_k < 6; eml_d_k++) {

  580       }
  581     }
  582   
  583     for (eml_i = 0; eml_i < 3; eml_i++) {
  584       eml_e_c = 6 * eml_i;
  585       for (eml_d_k = 6; eml_d_k > 0; eml_d_k += -1) {
  586         eml_f_c = (eml_d_k - 1) * 6;
  587         if (eml_Y[(eml_d_k + eml_e_c) - 1] != 0.0) {
